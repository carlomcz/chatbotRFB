def troca_acentos(txt): #retira os caracteres acentuados e substitui por caracteres sem acentos
  transTable = txt.maketrans("áàãéêíóôõúÁÃÃÉÊÍÓÔÕÚçÇ", "aaaeeiooouaaaeeioooucc")
  txt = txt.translate(transTable)
  return str(txt)

# função que tokeniza as palavras
#
def tokenizar(texto): 
#
    car_nok = ['SPACE', 'DET','ADP', 'CCONJ','PUNCT','PRON','SCONJ']
    txx = texto
    docxx = nlp(txx)
    txx_lemma = [(token.text,token.orth_, token.lemma_, token.pos_) for token in docxx]
    pal_xx = ''
    plvxx = ""
    for i in txx_lemma:
        if i[3] not in car_nok:
            plvxx = plvxx + ste.stem(i[0]) + ' '
    plvxx = plvxx.strip()
    plvxx = troca_acentos(plvxx)
    return plvxx

# funçao que calcula a similaridade do texto digitado com os serviços
#
def similaridade_servico(texto):

    # Carregando as dependencias do modelo 1
    dct_loaded1 = Dictionary.load('dct1')
    model_loaded1 = TfidfModel.load('model1')
    index_loaded1 = similarities.Similarity.load('index1')

    # Criando um texto de exemplo e tokenizando 1
    novo_texto_tokens = texto
    novo_texto_tokens = novo_texto_tokens.split()

    # Converte corpus para o formato BOW (Bag Of Words)
    novo_texto_corpus = dct_loaded1.doc2bow(novo_texto_tokens)

    # Buscando a similaridade
    sim = index_loaded1[model_loaded1[novo_texto_corpus]]

    #
    dfx1 = pd.DataFrame()
    dfx1['srv'] = df1['Servico']
    dfx1['lnk'] = df1['Link']
    dfx1['oQue'] = df1['oQue']
    dfx1['sim'] = sim
    #dfx1['Cat'] = df1['Categoria']
    dfx1 = dfx1.sort_values(by=['sim'], ascending=False)
    volta = ""
    for y in range(5):
        volta = volta + 'Serviço:' + dfx1.iloc[y][0] + "|Similaridade:" +  str(dfx1.iloc[y][3]) + '|Link:' + dfx1.iloc[y][1] + '<p>'
    return dfx1

def similaridade_conversa(texto):

    index_loaded2 = similarities.Similarity.load('index2')

    # Criando um texto de exemplo e tokenizando 1
    novo_texto_tokens = texto
    novo_texto_tokens = novo_texto_tokens.split()

    # Converte corpus para o formato BOW (Bag Of Words)
    novo_texto_corpus = dct_loaded2.doc2bow(novo_texto_tokens)

    # Buscando a similaridade
    sim = index_loaded2[model_loaded2[novo_texto_corpus]]

    #
    dfx2 = pd.DataFrame()
    dfx2['msg'] = df2['Pergunta']
    dfx2['resp'] = df2['Resposta']

    dfx2 = dfx2.sort_values(by=['sim'], ascending=False)
    volta = ""
    for y in range(3):
        volta = volta + 'Msg:' +  dfx2.iloc[y][0]+ '|Resposta:' +  dfx2.iloc[y][1] + "|Similaridade:" +  str(dfx2.iloc[y][2]) + '<p>'
    return dfx2
